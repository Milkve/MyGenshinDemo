LoginTipView = BaseClass(UINode)
LoginTipView.Singleton = true

function LoginTipView:Construtor(parent)
    self.viewCfg.prefabName = "UILoginTip"
    self.viewCfg.prefabPath = "ui/login/UILoginTip.prefab"
    self.viewCfg.isDestroyOnClose =true
    self.viewCfg.parent = parent
    self.viewCfg.component = {
        --渐隐
        {UI.Fade, {alpha = 0.9, durtion = 0.3, delay = 0.1}}, 
        --延迟操作
        {UI.DelayOperation, {func_callback = function()UIManager:Close(self)end,delay=2}}
    }
end

function LoginTipView:SetData(text)
    self.Text_Text.text = text
end

function LoginTipView:AddEvent()

end
